100  template <class FACTOR>
101  static bool ContainsCombinedNavStateKey(const FACTOR& factor, const int key_index) {
102  if (factor.find(sym::P(key_index)) != factor.end()) return true;
103  if (factor.find(sym::V(key_index)) != factor.end()) return true;
104  if (factor.find(sym::B(key_index)) != factor.end()) return true;
105  return false;
106  }
