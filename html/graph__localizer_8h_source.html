200  template <typename FactorType>
201  void DeleteFactors() {
202  int num_removed_factors = 0;
203  for (auto factor_it = graph_.begin(); factor_it != graph_.end();) {
204  if (dynamic_cast<FactorType*>(factor_it->get())) {
205  factor_it = graph_.erase(factor_it);
206  ++num_removed_factors;
207  continue;
208  }
209  ++factor_it;
210  }
211  LogDebug("DeleteFactors: Num removed factors: " << num_removed_factors);
212  }
